Exercice

Voici un exemple de code en language C permettant de remplir une matrice,calculer et afficher la somme de ses elements positifs: 

            
             include  <stdio.h>
            

             int  main() {
               int  lignes,colonnes,i,j,sum=0;
                  printf("Saisir le nombre de lignes et colonnes");
                  scanf("%d %d",&lignes,&colonnes);

                   int matrix[lignes][colonnes];
                  printf("Saisir les elements de la matrice");
                   for  (i = 0; i < lignes; i++) {
                     for  (j = 0; j < colonnes; j++) {
                         scanf("%d",%matrix[i][j]);
                          if (matrix[i][j]>0){
                         sum += matrix[i][j]; 
                      }
                    }
                  }
                       printf("Sum of positive elements: %d\n",sum); 
                        return  0; 
                      
              }
